# modbus with stm32f1

serial port: USART1  :   
pins : TX : PA9   
     : RX : PA10 

serial port: UART2  : console 
pins : TX : PA2   
     : RX : PA3	 

serial port: UART3  : primary sensor
pins : TX : PB10   
     : RX : PB11
